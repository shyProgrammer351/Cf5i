80점이상이면 끝내주게 잘 했다를 출력하고, 
70점이상이면 잘 했다를 출력하고, 
나머지는 했다를 출력한다.

	// 파일명 : ./Chapter05/ScoreMessage.c

	#include <stdio.h>

		 

	// 프로그램 시작 

1	int main() {

		int score = 0;

2		scanf( "%d", &score );

		 

3		switch ( score / 10 ) {

			// 90점이상이면 와! 끝내주게 잘 했다를 출력 

4			case 10:

5			case 9:

6				printf( "와! " );

			// 80점이상이면 끝내주게 잘 했다를 출력 

7			case 8:

8				printf( "끝내주게 " );

			// 70점이상이면 잘 했다를 출력 

9			case 7:

10				printf( "잘 " );

			// 나머지는 했다를 출력 

11			default:

12				printf( "했다" );

		}

		 

		// 프로그램 종료 

13		return 0;

	}
