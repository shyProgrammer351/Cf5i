문제

A양과 B군이 가위바위보를 선택하면 승패를 출력하는 문제입니다 실행순서를 클릭하세요 
A양, 가위(1), 바위(2), 보(3) 중 하나를 숫자로 입력하세요: 2
바위를 냈습니다.
B군, 가위(1), 바위(2), 보(3) 중 하나를 숫자로 입력하세요: 1
가위를 냈습니다.
A양이 이겼습니다.

프로그램 코드

	#include <stdio.h> 


1	int main() {

		int playerA = 0;

2		int playerB = 0;


3		printf( "A양, 가위(1), 바위(2), 보(3) 중 하나를 숫자로 입력하세요: " );

4		scanf( "%d", &playerA );


5		switch( playerA ) {

6		case 1:										

7			printf( "가위를 냈습니다.\n" ); 

8			break;									

9		case 2:										

10			printf( "바위를 냈습니다.\n" ); 

11			break;									

12		case 3:										

13			printf( "보를 냈습니다.\n" ); 

14			break;									

15		default:									

16			printf( "가위바위보를 잘못 냈습니다.\n" ); 

		}


17		printf("B군, 가위(1), 바위(2), 보(3) 중 하나를 숫자로 입력하세요: ");

18		scanf( "%d", &playerB );


19		switch( playerB ) {

20		case 1:										

21			printf( "가위를 냈습니다.\n" ); 	

22			break;									

23		case 2:										

24			printf( "바위를 냈습니다.\n" ); 

25			break;									

26		case 3:										

27			printf( "보를 냈습니다.\n" ); 

28			break;									

29		default:									

30			printf( "가위바위보를 잘못 냈습니다.\n" ); 

		}


31		if ( playerA == playerB ) {

32			printf( "\nA양과 B군이 비겼습니다.\n" ); 

		}

33		else if ( ( playerA == 1 ) && ( playerB == 3 ) ) {	

34			printf( "\nA양이 이겼습니다.\n" ); 

		}

35		else if ( ( playerA == 2 ) && ( playerB == 1 ) ) {

36			printf( "\nA양이 이겼습니다.\n" ); 

		}

37		else if ( ( playerA == 3 ) && ( playerB == 2 ) ) {

38			printf( "\nA양이 이겼습니다.\n" ); 

		}

39		else {

40			printf( "\nB군이 이겼습니다.\n" ); 

		}


41		return 0;

	}
