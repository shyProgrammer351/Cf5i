문제

컵 3개 중 하나에 임의로 공(O)을 숨기고 공의 위치를 맞추는 문제입니다 실행순서를 클릭하세요 
  ___    ___    ___  
 |   |  |   |  |   | 
 | 1 |  | 2 |  | 3 | 
   O 
 찾았다!
 1, 2, 3 중에서 공을 숨긴 컵을 찾으세요 : 1

프로그램 코드

	#include <stdio.h>

	#include <stdlib.h>

	#include <time.h>


1	int main() {

		srand( time( NULL ) );

2		int ball = rand() % 3 + 1;


3		printf( "  ___    ___    ___  \n" );

4		printf( " |   |  |   |  |   | \n" );

5		printf( " | 1 |  | 2 |  | 3 | \n\n\n\n" );

6		printf( "\n\n 1, 2, 3 중에서 공을 숨긴 컵을 찾으세요 : " );

		int cup = 0;

7		scanf( "%d", &cup );

			

8		printf( " \033[4;%dfO\n \033[5;%df", (ball*7-3), (cup*7-5) );

9		( cup == ball ) ?

10			printf( "찾았다!\n") :

11			printf( "놓쳤다!\n" );


12		return 0;

	}
