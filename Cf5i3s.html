문제

한 글자를 입력받아 그 값이 알파벳 소문자, 알파벳 대문자, 숫자, 기호 중 무엇인지를 출력하는 문제입니다 실행순서를 클릭하세요 
한 글자를 입력하세요: 2
숫자를 입력했군요^^

프로그램 코드

	#include <stdio.h>


1	int main() {

2		char letter = '\0';


3		printf("한 글자를 입력하세요: " );

4		scanf( "%c", &letter );


5		if (  ( 'a' <= letter ) && ( letter <= 'z' ) ) {

6			printf( "알파벳 소문자를 입력했군요^^" );

		}

7		else if (  ( 'A' <= letter ) && ( letter <= 'Z' ) ) {

8			printf( "알파벳 대문자를 입력했군요^^" );

		}

9		else if (  ( '0' <= letter ) && ( letter <= '9' ) ) {

10			printf( "숫자를 입력했군요^^" );

		}

11		else {

12			printf( "기호를 입력했군요^^" );

		}

		

13		return 0;

	}
